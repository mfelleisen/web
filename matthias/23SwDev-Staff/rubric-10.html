    ;; A configuration is a functional, String-keyed dictionary

    ;; with a fixed set of keys.

    

    #; {String Any ... String Any -> Configuration}

    #; (make-configuration key1 value1 ...)

    ;; creates a configuration from the given keys and default values

    (define (configuration . key-value-pairs) ...)

    

    ;; to be precise, the above "function" generates the functions below per

    ;; specification of a configuration and it enforces that only the

    ;; keys specified for the default config are usable

    

    #; {JSexpr [Listof String] -> Configuration}

    #; (jsexpr->configuration j [list s ...])

    ;; consumes the data rep of a JSON object and creates a configuration

    ;; with the given keys (and only those)

    (define (jsexpr->configuration j lok) ...)

    

    #; {Configuration String -> Any}

    #; (config-ref c k)

    ;; retrieves the current value of key `k` in configuration `c`

    ;; EFFECT errors if `k` isn't available

    (define (config-ref c k) ..)

    

    #; {Configuration String Any -> Configuration}

    #; (config-set c k v)

    ;; creates a configuration like the old one,

    ;; with the value of key `k` set to `v` and otherwise the same

    ;; EFFECT errors if `k` isn't available

    (define (config-set c k v) ..)

    

    

    ;; EXAMPLE

    (configuration referee

      [STATE0 #false

        #:to-jsexpr state->jsexpr

        #:from-jsexpr jsexpr->state

        #:is-a "JState"]

      [QUIET #true

        #:is-a "Boolean"]

      [PER-TURN 6

        #:is-a "Natural" "less than or equal to 6"]

      [OBSERVE void

        #:to-jsexpr   (λ (x) (not (eq? x void)))

        #:from-jsexpr (λ (x) (if x #t void))

        #:is-a "Boolean"]

      [CONFIG-S DEFAULT-CONFIG-S

        #:to-jsexpr refereeState-config->jsexpr

        #:from-jsexpr jsexpr->refereeState-config

        #:is-a "RefereeStateConfig"])
