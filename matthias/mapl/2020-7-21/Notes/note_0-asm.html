1.1 Finite State Machines

Return to Systematic Program Design: The Idea in the book.

  #! /usr/bin/env racket
  #lang racket 
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - - 
   
  ;; The program reads input chars one at a time from
  ;; Standard Input (StdIn) until it recognizes the word
  ;; "raise" or until it knows that the word isn't "raise".
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - - 
  ;; DATA DEFINITIONS
   
  (define FAILURE "failure")
   
  ;; A State is one of:
  ;; - ""
  ;; - "r"
  ;; - "ra"
  ;; - "rai"
  ;; - "rais"
  ;; - "raise"
  ;; - FAILURE
  ;; The InitialState is: "".
  ;; A FinalState is one of: "raise" or FAILURE.
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - -
  ;; CODE 
   
  #; {-> String}
  (define (driver)
    (define initial-state (load))
    (define final-state 
      (let while ([current initial-state])
        (cond
          [(final-state? current) current]
          [else
           (define next-char (read-char))
           (while (transition current next-char))])))
    (define out (unload final-state))
    out)
   
  #; { -> InitialState}
  (define (load)
    "")
   
  #; {State Char -> State}
  ;; determine the next state from `current` and `input-char`
  ;; ASSUME `current` is _not_ a final state. 
  (define (transition current next-char)
    (case current
      [("") (if (equal? next-char #\r) "r" FAILURE)]
      [("r") (if (equal? next-char #\a) "ra" FAILURE)]
      [("ra") (if (equal? next-char #\i) "rai" FAILURE)]
      [("rai") (if (equal? next-char #\s) "rais" FAILURE)]
      [("rais") (if (equal? next-char #\e) "raise" FAILURE)]))
   
  #; {State -> Boolean}
  (define (final-state? s)
    (or (equal? s FAILURE) (equal? s "raise")))
   
  #; {FInalState -> String}
  (define (unload s)
    (if (equal? s "raise")
        "the machine recognized \"raise\""
        "the machine did NOT recognize \"raise\""))
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - -
  ;; TESTS
   
  (module+ test
    (require rackunit)
   
    (check-equal? (transition "r" #\a) "ra")
    (check-equal? (transition "r" "i") FAILURE))
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - -
  ;; RUN PROGRAM RUN
   
  (module+ main
    (driver))

1.2 An Infinite State Machine

  #! /usr/bin/env racket
  #lang racket 
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - - 
   
  ;; The program implements an abstract state machine.
  ;; It reads input lines one at a time from StdIn, expecting
  ;; the n-th line to contain the number n.
  ;; 
  ;; If there is no line to read witin INTERVAL s or if the
  ;; input is something other than n, the machine fails.
  ;;
  ;; There is no successful termination. 
   
  (define INTERVAL 2)
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - - 
  ;; DATA DEFINITIONS
   
  ;; A State is either an instance of
  ;; - InitialState 
  ;; - GoodState
  ;; - FinalState
   
  (define good-state%
    (class* object% (equal<%>)
      (init-field n) 
      (super-new)
      
      (define/public (transition k)
        (if (natural? k)
            (if (= k (+ n 1)) (new good-state% [n k]) FAILURE)
            FAILURE))
   
      (define/public (is-final-state?)
        #false)
   
      ;; object equality 
      (define/public (equal-to? other recur)
        (= n (get-field n other)))
      (define/public (equal-hash-code-of hash-code) 10)
      (define/public (equal-secondary-hash-code-of hc) 100)))
   
  (define FAILURE ;; singleton 
    (new
     (class object%
      (super-new)
      (define/public (is-final-state?) #true)
      (define/public (unload) "failure"))))
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - -
  ;; CODE 
   
  #; {-> String}
  (define (driver)
    (define initial-state (load))
    (define final-state 
      (let while ([current initial-state])
        (cond
          [(send current is-final-state?) current]
          [else
           (define k (timed-read string->number INTERVAL))
           (while (send current transition k))])))
    (define out (send final-state unload))
    out)
   
  (define (load) 
    (new good-state% [n 0]))
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - -
  ;; LIBRARY
   
  #; {(∀ (x) (-> (-> String x) Positive-Real (U False x)))}
  (define (timed-read string->desired interval)
    (define nxt (sync/timeout interval (ready)))
    (cond
      [(boolean? nxt)        #false]
      [(eof-object? nxt)     #false]
      [(string->desired nxt) => identity]
      [else                  #false]))
   
  (define (ready)
    (read-line-evt (current-input-port) 'linefeed))
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - -
  ;; TESTS
   
  (module+ test
    (require rackunit)
   
    (check-equal? (send (new good-state% [n 0]) transition 1)
                  (new good-state% [n 1])))
   
  ;; - - - - - - - - - - - - - - - - - - - - - - - - - - -
  ;; RUN PROGRAM RUN
   
  (module+ main
    (driver))
