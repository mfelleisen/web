2 Classes

#lang scheme
(define c%                  ; name the class with a normal define
  (class object%            ; the superclass is object%
         (init-field x)     ; y is a constructor argument and a field
         (define/public (m) ; m is a one-argument method (this)
           (new c% [x 0]))  ; and it returns a new instance of c%
         (super-new)))      ; let's initialize it all the way to the top
